class DeepSpeedTrainer:
    """
    DeepSpeed 训练器

    参考 Day011 pretrain.py 的实现方式
    支持 ZeRO 优化、混合精度训练、梯度累积
    """

    def __init__(self, args):
        self.args = args

        # 初始化分布式环境（与 Day011 pretrain.py 一致）
        deepspeed.init_distributed()

        # 获取 rank
        self.rank = dist.get_rank()
        self.world_size = dist.get_world_size()
        self.is_main_process = (self.rank == 0)

        # 初始化日志（仅主进程）
        if self.is_main_process and HAS_TENSORBOARD:
            self.log = SummaryWriter(f"runs/{datetime.now().strftime('%Y%m%d_%H%M%S')}")
        else:
            self.log = None

        # 加载 DeepSpeed 配置
        self._load_deepspeed_config()

        # 创建模型
        self._create_model()

        # 初始化 DeepSpeed
        self._init_deepspeed()

        # 训练状态
        self.global_step = 0

    def _load_deepspeed_config(self):
        """加载并动态调整 DeepSpeed 配置"""
        if self.is_main_process:
            with open(self.args.deepspeed_config, 'r') as f:
                self.ds_config = json.load(f)

            print(f"DeepSpeed Config loaded")
        else:
            self.ds_config = None

    def _create_model(self):
        """创建模型"""
        config_map = {
            'mini': ConfigFactory.create_mini_model,
            'small': ConfigFactory.create_small_model,
            'vlm': ConfigFactory.create_vlm_model
        }

        model_config = config_map[self.args.model_size]()

        if model_config.use_vision:
            self.model = LlamaForConditionalGeneration(model_config)
        else:
            self.model = LlamaForCausalLM(model_config)

        if self.is_main_process:
            num_params = sum(p.numel() for p in self.model.parameters())
            print(f"Model created: {num_params:,} parameters")

    def _init_deepspeed(self):
        """初始化 DeepSpeed 引擎"""
        # 创建数据集
        if self.args.data_file and os.path.exists(self.args.data_file):
            from dataset import PretrainDataset
            train_dataset = PretrainDataset(
                self.args.data_file,
                self.args.paragraph_size
            )
        else:
            model_config = ConfigFactory.create_mini_model()
            train_dataset = RandomTextDataset(
                num_samples=self.args.num_samples,
                seq_length=self.args.paragraph_size,
                vocab_size=model_config.vocab_size
            )

        # 损失函数（忽略 padding token）
        self.loss_fn = nn.CrossEntropyLoss(ignore_index=0)

        # 初始化 DeepSpeed（与 Day011 pretrain.py 一致）
        model_parameters = list(self.model.parameters())

        self.engine, self.optimizer, _, self.lr_scheduler = deepspeed.initialize(
            args=self.args,
            model=self.model,
            model_parameters=model_parameters,
            training_data=train_dataset
        )

        if self.is_main_process:
            print(f"DeepSpeed initialized: {self.world_size} GPUs")

    def train(self):
        """主训练循环"""
        self.engine.train()

        # 尝试加载检查点
        _, client_sd = self.engine.load_checkpoint(
            self.args.checkpoint_dir,
            tag=None
        )

        if client_sd is None:
            client_sd = {"step": 0}

        self.global_step = client_sd['step']

        # 训练循环
        for i, batch in enumerate(self.engine.training_dataloader()):
            # 确保数据在正确设备上
            input_ids = batch['input_ids'].to(device=self.engine.device, dtype=torch.long)

            # 因果推理：xs 是输入，ys 是目标
            xs = input_ids[:, :-1]
            ys = input_ids[:, 1:]

            # 前向传播
            outputs = self.engine(xs)

            # 处理输出
            if isinstance(outputs, dict):
                logits = outputs.get('logits', outputs.get(0))
            else:
                logits = outputs

            # 转换 logits 形状
            logits = logits.reshape(-1, logits.size(-1))

            # 计算损失
            ys = ys.reshape(-1)
            loss = self.loss_fn(logits, ys)

            # DeepSpeed 反向传播（与 Day011 pretrain.py 一致）
            self.engine.backward(loss)

            # DeepSpeed 优化步骤
            self.engine.step()

            # 日志（仅主进程）
            if self.is_main_process and i % 100 == 0:
                print(f"Step {self.global_step} | Loss: {loss.item():.4f}")
                if self.log and HAS_TENSORBOARD:
                    self.log.add_scalar("loss", loss.item(), self.global_step)

            self.global_step += 1

        # 保存检查点
        self.save_checkpoint()

    def save_checkpoint(self):
        """保存检查点"""
        save_tag = self.args.save_tag

        self.engine.save_checkpoint(
            self.args.checkpoint_dir,
            tag=f"{save_tag}",
            client_state={"step": self.global_step}
        )

        if self.is_main_process:
            print(f"Checkpoint saved to {self.args.checkpoint_dir}/{save_tag}")

    def cleanup(self):
        """清理资源"""
        if self.log:
            self.log.close()

class DeepSpeedTrainerSimple:
    """
    简化的 DeepSpeed 训练器

    使用标准 PyTorch DataLoader
    """

    def __init__(self, args):
        self.args = args

        # 初始化分布式环境
        deepspeed.init_distributed()

        # 获取 rank
        self.rank = dist.get_rank()
        self.world_size = dist.get_world_size()
        self.is_main_process = (self.rank == 0)

        # 创建模型
        model_config = ConfigFactory.create_mini_model()
        self.model = LlamaForCausalLM(model_config)

        # 创建 DataLoader
        dataset = RandomTextDataset(
            num_samples=args.num_samples,
            seq_length=args.paragraph_size,
            vocab_size=model_config.vocab_size
        )
        self.train_loader = DataLoader(
            dataset,
            batch_size=1,
            shuffle=True
        )

        # 初始化 DeepSpeed
        from torch.optim import AdamW
        optimizer = AdamW(self.model.parameters(), lr=1e-4)

        self.engine, _, _, _ = deepspeed.initialize(
            args=args,
            model=self.model,
            optimizer=optimizer
        )

        # 损失函数
        self.loss_fn = nn.CrossEntropyLoss(ignore_index=0)

        # 训练状态
        self.global_step = 0

    def train(self):
        """训练循环"""
        self.engine.train()

        for batch in self.train_loader:
            input_ids = batch['input_ids'].to(self.engine.device)

            xs = input_ids[:, :-1]
            ys = input_ids[:, 1:]

            outputs = self.engine(xs)

            if isinstance(outputs, dict):
                logits = outputs['logits']
            else:
                logits = outputs

            logits = logits.reshape(-1, logits.size(-1))
            ys = ys.reshape(-1)
            loss = self.loss_fn(logits, ys)

            self.engine.backward(loss)
            self.engine.step()

            if self.is_main_process and self.global_step % 100 == 0:
                print(f"Step {self.global_step} | Loss: {loss.item():.4f}")

            self.global_step += 1

    def save(self, save_dir):
        """保存检查点"""
        self.engine.save_checkpoint(save_dir)

            vocab_size=self.vocab_size
        )

        # 初始化 DeepSpeed（与 Day011 pretrain.py 一致）
        self.engine, self.opt, self.training_dataloader, self.lr_scheduler = deepspeed.initialize(
            args=self.args,
            model=self.model,
            training_data=dataset,
            model_parameters=self.model.parameters(),
            config="./deepspeed_config.json",
        )

        # 损失函数忽略 pad token (id=0)
        self.loss_fn = nn.CrossEntropyLoss(ignore_index=0)

    def __call__(self):
        rank = deepspeed.comm.get_rank()
        self.engine.train()

        # 加载检查点（如果存在）
        _, client_sd = self.engine.load_checkpoint("weights")
        if client_sd is None:
            client_sd = {"step": 0}

        max_steps = self.args.max_steps

        for i, batch in enumerate(self.training_dataloader):
            # 处理 Day013 RandomTextDataset 返回的 dict
            if isinstance(batch, dict):
                input_ids = batch['input_ids']
            else:
                input_ids = batch

            ds = input_ids.to(device=self.engine.device, dtype=torch.long)

            # 因果推理：xs 是输入，ys 是目标
            xs = ds[:, :-1]
            ys = ds[:, 1:]

            # 前向传播
            outputs = self.engine(xs)

            # 提取 logits（LlamaForCausalLM 返回 dict）
            if isinstance(outputs, dict):
                logits = outputs['logits']
            else:
                logits = outputs

            # 转换形状 [B, seq_len, vocab_size] -> [seq_len * B, vocab_size]
            logits = logits.reshape(-1, self.vocab_size)
            ys = ys.reshape(-1)

            # 计算损失
            loss = self.loss_fn(logits, ys)

            # DeepSpeed 反向传播和优化
            self.engine.backward(loss)
            self.engine.step()

            step = client_sd["step"]
            if rank == 0:
                if i % 10 == 0:
                    print(f"Step {step} | Loss: {loss.item():.4f}")
                    if self.log:
                        self.log.add_scalar("loss", loss.item(), step)

            client_sd["step"] += 1

            # 限制最大步数
            if max_steps is not None and i >= max_steps - 1:
                break

        # 保存检查点
        save_tag = self.args.ss
        self.engine.save_checkpoint("weights", tag=f"{save_tag}", client_state={"step": client_sd['step']})


if __name__ == '__main__':
    train = Trainer()
    train()
